
country_decisions = {
	
	start_demonsterization = {
		potential = {
			has_country_modifier = monstrous_nation
			NOT = { has_country_flag = removing_monstrous }
			NOT = { has_country_flag = no_longer_monstrous }
			NOT = { has_ruler_flag = undergoing_xhazob_ritual }
		}
		allow = {
			hidden_trigger = {
				has_country_modifier = monstrous_nation
			}
			#NOT = { current_age = age_of_discovery }
			adm_tech = 7
			stability = 1

			any_known_country = {
				NOT = { has_country_modifier = monstrous_nation }
			}
			if = {
				limit = { ruler_has_personality = xhazobs_chosen_personality }
				NOT = { ruler_has_personality = xhazobs_chosen_personality }
			}
		}
		effect = {
			if = {
				limit = { ai = no }
				add_stability = -1
			}
			hidden_effect = { set_country_flag = removing_monstrous }
			hidden_effect = { 
				if = {
					limit = { 
						has_country_flag = semi_monstrous 
						NOT = { has_country_flag = passively_removing_monstrous }
					}
					set_variable = { monstrousPoints = 50 }
				}
				else_if = { 
					limit = { NOT = { has_country_flag = passively_removing_monstrous } }
					set_variable = { monstrousPoints = 100 } 
				}
				else = {
					clr_country_flag = passively_removing_monstrous
				}
			}

			custom_tooltip = start_demonsterization
		}
		ai_will_do = { 
			factor = 400 
			modifier = {
				factor = 0
				OR = {
					ruler_has_personality = cruel_personality
					ruler_has_personality = malevolent_personality
					ruler_has_personality = zealot_personality
					NOT = { want_to_decrease_monstrous = yes }
				}
			}
		}
	}
	
	start_passive_demonsterization = {
		potential = {
			has_country_modifier = monstrous_nation
			NOT = { has_country_flag = passively_removing_monstrous }
			NOT = { has_country_flag = removing_monstrous }
			NOT = { has_country_flag = no_longer_monstrous }
		}
		allow = {
			hidden_trigger = {
				has_country_modifier = monstrous_nation
			}
			#NOT = { current_age = age_of_discovery }
			has_institution = feudalism
			stability = 0

			any_known_country = {
				NOT = { has_country_modifier = monstrous_nation }
			}
			
			any_owned_province = {
				OR = {
					AND = {
						has_owner_culture = yes
						has_dwarven_minority_trigger = yes
					}
					AND = {
						has_owner_culture = yes
						has_elven_minority_trigger = yes
					}
					AND = {
						has_owner_culture = yes
						has_gnomish_minority_trigger = yes
					}
					AND = {
						has_owner_culture = yes
						has_half_elven_minority_trigger = yes
					}
					AND = {
						has_owner_culture = yes
						has_halfling_minority_trigger = yes
					}
					AND = {
						has_owner_culture = yes
						has_harimari_minority_trigger = yes
					}
					AND = {
						has_owner_culture = yes
						has_human_minority_trigger = yes
					}
					has_dwarven_majority_trigger = yes
					has_elven_majority_trigger = yes
					has_gnomish_majority_trigger = yes
					has_half_elven_majority_trigger = yes
					has_halfling_majority_trigger = yes
					has_harimari_majority_trigger = yes
					has_human_majority_trigger = yes
				}
			}
			OR = {
				medium_tolerance_dwarven_race_trigger = yes
				high_tolerance_dwarven_race_trigger = yes
				
				medium_tolerance_elven_race_trigger = yes
				high_tolerance_elven_race_trigger = yes
				
				medium_tolerance_gnomish_race_trigger = yes
				high_tolerance_gnomish_race_trigger = yes
				
				medium_tolerance_half_elven_race_trigger = yes
				high_tolerance_half_elven_race_trigger = yes
				
				medium_tolerance_halfling_race_trigger = yes
				high_tolerance_halfling_race_trigger = yes
				
				medium_tolerance_harimari_race_trigger = yes
				high_tolerance_harimari_race_trigger = yes
				
				medium_tolerance_human_race_trigger = yes
				high_tolerance_human_race_trigger = yes
			}
		}
		effect = {
			hidden_effect = { set_country_flag = passively_removing_monstrous }
			hidden_effect = { 
				if = {
					limit = { has_country_flag = semi_monstrous }
					set_variable = { monstrousPoints = 50 }
				}
				else = { set_variable = { monstrousPoints = 100 } }
			}

			custom_tooltip = start_passive_demonsterization
		}
		ai_will_do = { 
			factor = 400 
			modifier = {
				factor = 0
				OR = {
					ruler_has_personality = cruel_personality
					ruler_has_personality = malevolent_personality
					ruler_has_personality = zealot_personality
					NOT = { want_to_decrease_monstrous = yes }
				}
			}
		}
	}
	
	current_monstrous_level = {
		potential = {
			ai = no
			has_country_modifier = monstrous_nation
			OR = {
				has_country_flag = passively_removing_monstrous
				has_country_flag = removing_monstrous
			}
		}
		
		allow = {
			always = no
		}
		
		effect = {
			custom_tooltip = current_monstrous_level_tooltip
		}
	}
}
